# 核心API路由（app/api/chat/route.ts）
import OpenAI from 'openai';
import { NextResponse } from 'next/server';

const openai = new OpenAI();

export async function POST(req: Request) {
  const { message, tier } = await req.json();
  
  // 根据订阅层级选择模型
  const model = tier === 'pro' ? 'gpt-4o' : 'gpt-4o-mini';
  
  const completion = await openai.chat.completions.create({
    model,
    messages: [{ role: 'user', content: message }],
    max_tokens: 2000,
  });
  
  return NextResponse.json({
    reply: completion.choices[0].message.content,
    usage: completion.usage,
  });
}

这段代码的核心思路：用同一套逻辑服务不同层级的用户，Pro用户用更好的模型，Free用户用轻量模型。成本差异巨大——gpt-4o-mini每1000次调用大约$0.15，而gpt-4o要$2.50。这就是为什么分层定价不只是营销策略，更是成本控制的生命线。

第三步：定价的心理学与数学

我犯过的最大错误？定价太低。起初我定$9/月，觉得"便宜点用户会来"。结果每月收入$450，服务器成本$200，净利润仅$250。后来我涨价到$29/月，用户数只掉了30%，但收入变成了$1,020。

定价公式我后来总结成这样：

你的月定价 ≥ (你的API成本 × 10) + $10平台费 + 你的时薪价值

举例：如果每次请求成本$0.01，用户月均用1000次，你的API成本是$10。那定价至少是$10×10 + $10 = $110/月。听起来贵？不，对于B2B工具这个价格合理得惊人。

案例一：AI简历优化器 —— 我的第一桶金

产品：用户上传简历PDF，AI分析并给出优化建议，支持ATS关键词匹配。

技术实现：核心就是把PDF转文本，然后喂给Claude：

# pdf_processor.py
import anthropic
from PyPDF2 import PdfReader

def analyze_resume(pdf_path: str, job_description: str) -> dict:
    reader = PdfReader(pdf_path)
    text = "\n".join(page.extract_text() for page in reader.pages)
    
    client = anthropic.Anthropic()
    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=4000,
        messages=[{
            "role": "user",
            "content": f"""分析这份简历与目标职位的匹配度：
            
简历内容：
{text}

目标职位描述：
{job_description}

请提供：
1. 匹配度评分(0-100)
2. 缺失的关键词(前5个)
3. 具体修改建议(3条)
4. 优化后的简历片段示例"""
        }]
    )
    return {"analysis": response.content[0].text}

案例二：自动Newsletter —— 真正的"睡后收入"

产品：AI自动抓取特定领域新闻，生成分析newsletter，通过Beehiiv分发。

关键架构：这套系统运行了8个月，我几乎没手动干预过：

# newsletter_pipeline.py — 每日凌晨3点自动运行
import schedule
from datetime import datetime

def daily_pipeline():
    # 1. 抓取RSS源（50+个信息源）
    articles = fetch_rss_feeds(SOURCES)
    
    # 2. AI筛选：只保留高价值内容
    filtered = ai_filter(
        articles,
        criteria="与AI创业、融资、产品发布相关，排除纯营销内容"
    )
    
    # 3. AI生成深度分析（不是简单摘要）
    analysis = generate_deep_analysis(
        filtered[:5],  # 只取top5
        style="犀利、有观点、带个人判断"
    )
    
    # 4. 格式化并发布到Beehiiv
    publish_to_beehiiv(analysis, segment="premium")
    
    # 5. 同步发布到web（SEO价值）

# 定时任务配置
schedule.every().day.at("03:00").do(daily_pipeline)

变现模式：免费版3篇/周，付费版$12/月获取每日内容+独家分析。目前有800+付费订阅者。

⚠️ 这个模式的陷阱

看似完美，但有两个隐性成本你必须知道：

  • 内容质量维护：AI生成的内容偶尔会"幻觉"——编造不存在的新闻。我为此加了事实核查层，增加了30%的API成本。
  • 退订率：newsletter的月退订率约3-5%，你必须持续获取新订阅者才能维持增长。
